defined_text = { # Country/leader/pop/species scope
name = PlusZeroOrBlankOnDecadenceRateOfChangeMonthly
	text = {
		trigger = {
			check_variable = {
				which = redrising_decadence_rate_of_change_monthly
				value = 0
			}
		}
		localization_key = REDRISING_zero
	}
	text = {
		trigger = {
			check_variable = {
				which = redrising_decadence_rate_of_change_monthly
				value > 0.001
			}
		}
		localization_key = REDRISING_plus
	}
	text = {	
		trigger = {
			check_variable = {
				which = redrising_decadence_rate_of_change_monthly
				value < -0.001
			}
		}	
		localization_key = REDRISING_blank
	}
}
